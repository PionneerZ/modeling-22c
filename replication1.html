class GrayScaleModel:
    def __init__(self, initial_cash=1000, params=None):
        self.cash = initial_cash
        self.portfolio_value = []  # 
        
        # : {'BTC': {'amount': 0, 'avg_cost': 0, 'buy_date': None}, ...}
        self.holdings = {
            'BTC': {'amount': 0, 'avg_cost': 0, 'last_buy_date_idx': -999},
            'Gold': {'amount': 0, 'avg_cost': 0, 'last_buy_date_idx': -999}
        }
        
        #  ( Figure 6  [12, 0.6, 0.89])
        # T: , N: , E: 
        self.params = params if params else {
            'MA_window': 5,      #  ()
            'T_hold': 12,        # Section 3.7:  z
            'Profit_Margin': 10, # Section 3.4:  L ()
            'Fee_Rate': 0.02,    #  f (2%)
            'C_weight': 100000,  # Section 3.9:  C
            'Revert_Thresh': 0.6 # Section 3.2:  MA  60%  (Figure 6  N)
        }

    def calculate_ma_gradient(self, price_series, current_idx, n):
        """ (Section 3.1)"""
        if current_idx < n: return None, None
        
        #  n 
        past_prices = price_series.iloc[current_idx-n : current_idx]
        
        # Section 3.1: Moving Average A_t(n)
        ma = past_prices.mean() 
        
        # Section 3.1: Gradient G = P_t - P_{t-n+1}
        #  - n
        gradient = price_series.iloc[current_idx] - price_series.iloc[current_idx-n+1]
        
        return ma, gradient

    def get_time_weight(self, t):
        """Section 3.9:  W = C / t^2"""
        if t == 0: return 999999 # 0
        return self.params['C_weight'] / (t**2)

    def check_sell_signal(self, asset_name, current_price, current_day_idx):
        """Section 3.4 & 3.7: """
        holding = self.holdings[asset_name]
        
        if holding['amount'] == 0:
            return False, 0 # 
            
        # Section 3.7: Holding Period 
        #  z 
        days_held = current_day_idx - holding['last_buy_date_idx']
        if days_held < self.params['T_hold']:
            return False, 0
            
        # Section 3.4: 
        # S = (1 - f) * P - AvgCost - L
        fee = self.params['Fee_Rate']
        margin_L = self.params['Profit_Margin']
        
        # 
        net_revenue_per_unit = (1 - fee) * current_price
        #  ()
        profit_per_unit = net_revenue_per_unit - holding['avg_cost']
        
        #  S > 0 ( > L/amount)
        #  L 
        if profit_per_unit > margin_L:
            return True, profit_per_unit
        
        return False, 0

    def check_buy_signal(self, asset_name, current_price, ma, gradient, t_days):
        """Section 3.1 & 3.2: """
        signal_score = 0
        
        # Section 3.1: Momentum Trading
        # Buy if P > MA and Gradient > 0 (0)
        if current_price > ma and gradient > 0:
            # Section 3.3: 
            # Section 3.9:  1/t^2
            signal_score += gradient * self.get_time_weight(t_days)
            
        # Section 3.2: Reversion-to-Mean
        # Buy if P < MA * Threshold (Figure 6  N)
        if current_price < ma * self.params['Revert_Thresh']:
            # Section 3.3:  (MA - P)^2
            dist = (ma - current_price)
            signal_score += (dist ** 2) * self.get_time_weight(t_days)
            
        return signal_score

    def execute_trade(self, action, asset_name, price, current_day_idx, score=0):
        """"""
        fee = self.params['Fee_Rate']
        
        if action == 'SELL':
            amount = self.holdings[asset_name]['amount']
            revenue = amount * price * (1 - fee)
            
            self.cash += revenue
            # 
            self.holdings[asset_name]['amount'] = 0
            self.holdings[asset_name]['avg_cost'] = 0
            print(f"Day {current_day_idx}: SELL {asset_name} @ {price:.2f}. Cash: {self.cash:.2f}")
            
        elif action == 'BUY':
            #  50%
            invest_cash = self.cash * 0.5 
            if invest_cash < 10: return # 
            
            units = invest_cash / (price * (1 + fee))
            cost = price * (1 + fee)
            
            #  (Section 3.4 )
            old_amt = self.holdings[asset_name]['amount']
            old_cost = self.holdings[asset_name]['avg_cost']
            
            new_amt = old_amt + units
            new_avg_cost = (old_amt * old_cost + units * cost) / new_amt
            
            self.holdings[asset_name]['amount'] = new_amt
            self.holdings[asset_name]['avg_cost'] = new_avg_cost
            self.holdings[asset_name]['last_buy_date_idx'] = current_day_idx
            
            self.cash -= invest_cash
            print(f"Day {current_day_idx}: BUY {asset_name} @ {price:.2f} (Score: {score:.1f}). Cash: {self.cash:.2f}")

trader = GrayScaleModel(initial_cash=10000)

prices_btc = df['Price_BTC']
prices_gold = df['Price_Gold']
ma_window = trader.params['MA_window']

print("---  ---")

#  ( Section 3.8  Daily Loop)
# t  ma_window 
for t in range(ma_window, len(df)):
    
    # 1. 
    p_btc = prices_btc.iloc[t]
    p_gold = prices_gold.iloc[t]
    
    #  MA  
    ma_btc, grad_btc = trader.calculate_ma_gradient(prices_btc, t, ma_window)
    ma_gold, grad_gold = trader.calculate_ma_gradient(prices_gold, t, ma_window)
    
    # --- Section 3.8:  (Sell First) ---
    btc_sell, _ = trader.check_sell_signal('BTC', p_btc, t)
    gold_sell, _ = trader.check_sell_signal('Gold', p_gold, t)
    
    if btc_sell: trader.execute_trade('SELL', 'BTC', p_btc, t)
    if gold_sell: trader.execute_trade('SELL', 'Gold', p_gold, t)
        
    # --- Section 3.8:  (Buy Later) ---
    # 
    score_btc = trader.check_buy_signal('BTC', p_btc, ma_btc, grad_btc, t)
    score_gold = trader.check_buy_signal('Gold', p_gold, ma_gold, grad_gold, t)
    
    #  ( Compare)
    # 
    if score_btc > 100: # 
        trader.execute_trade('BUY', 'BTC', p_btc, t, score_btc)
    
    if score_gold > 100:
        trader.execute_trade('BUY', 'Gold', p_gold, t, score_gold)
        
    # 
    equity = trader.cash + \
             trader.holdings['BTC']['amount'] * p_btc + \
             trader.holdings['Gold']['amount'] * p_gold
    trader.portfolio_value.append(equity)

print("---  ---")
print(f": {trader.portfolio_value[-1]:.2f}")

# 
plt.plot(trader.portfolio_value)
plt.title("Portfolio Value Over Time")
plt.xlabel("Days")
plt.ylabel("Value ($)")
plt.show()
